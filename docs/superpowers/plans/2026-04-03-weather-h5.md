```typescript
export async function getCityId(location: string) {
  const key = process.env.QWEATHER_KEY;
  const res = await fetch(`https://geoapi.qweather.com/v2/city/lookup?location=${location}&key=${key}`);
  const data = await res.json();
  if (!data.location || data.location.length === 0) return null;
  return { id: data.location[0].id, name: data.location[0].name };
}

export async function getWeather7D(locationId: string) {
  const key = process.env.QWEATHER_KEY;
  const res = await fetch(`https://api.qweather.com/v7/weather/7d?location=${locationId}&key=${key}`);
  return res.json();
}

export async function getWarnings(locationId: string) {
  const key = process.env.QWEATHER_KEY;
  const res = await fetch(`https://api.qweather.com/v7/warning/now?location=${locationId}&key=${key}`);
  return res.json();
}
```

```typescript
import { kv } from '@vercel/kv';

export interface Subscriber {
  sendKey: string;
  locationId: string;
  cityName: string;
}

export async function upsertSubscriber(userId: string, data: Subscriber) {
  await kv.hset('subscribers', { [userId]: data });
}

export async function deleteSubscriber(userId: string) {
  await kv.hdel('subscribers', userId);
}

export async function getSubscribers(): Promise<Record<string, Subscriber>> {
  const data = await kv.hgetall('subscribers');
  return (data as Record<string, Subscriber>) || {};
}
```

```typescript
export function evaluateWeatherCondition(daily: any[], warnings: any[]) {
  if (warnings && warnings.length > 0) {
    return { shouldNotify: true, message: `气象预警: ${warnings[0].title}` };
  }

  if (daily.length >= 2) {
    const today = daily[0];
    const tomorrow = daily[1];

    const maxDrop = parseInt(today.tempMax) - parseInt(tomorrow.tempMax);
    const minDrop = parseInt(today.tempMin) - parseInt(tomorrow.tempMin);

    if (maxDrop >= 8 || minDrop >= 8) {
      return { shouldNotify: true, message: `温度骤降提醒！明天温度将大幅下降。` };
    }

    const rainTexts = ['雨', '雪'];
    const todayRain = rainTexts.some(t => today.textDay.includes(t));
    const tomorrowRain = rainTexts.some(t => tomorrow.textDay.includes(t));

    if (!todayRain && tomorrowRain) {
      return { shouldNotify: true, message: `变天提醒！明天有降水（${tomorrow.textDay}），请注意防范。` };
    }
  }

  return { shouldNotify: false, message: '' };
}
```

```typescript
import { NextResponse } from 'next/server';
import { getSubscribers } from '@/lib/kv';
import { getWeather7D, getWarnings } from '@/lib/qweather';
import { evaluateWeatherCondition } from '@/lib/weather-engine';
import { sendWechatPush } from '@/lib/serverchan';

export async function GET(request: Request) {
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return new NextResponse('Unauthorized', { status: 401 });
  }

  const subscribers = await getSubscribers();
  if (!subscribers) return NextResponse.json({ processed: 0 });

  const byLocation: Record<string, typeof subscribers[string][]> = {};
  for (const sub of Object.values(subscribers)) {
    if (!byLocation[sub.locationId]) byLocation[sub.locationId] = [];
    byLocation[sub.locationId].push(sub);
  }

  let sent = 0;
  for (const [locationId, users] of Object.entries(byLocation)) {
    const [weatherRes, warningRes] = await Promise.all([
      getWeather7D(locationId),
      getWarnings(locationId)
    ]);

    if (weatherRes?.daily) {
      const condition = evaluateWeatherCondition(weatherRes.daily, warningRes?.warning || []);
      if (condition.shouldNotify) {
        for (const user of users) {
          await sendWechatPush(user.sendKey, '【天气预警】' + user.cityName, condition.message);
          sent++;
        }
      }
    }
  }

  return NextResponse.json({ success: true, sent });
}
```

```tsx
"use client";
import { useState, useEffect } from 'react';
import { v4 as uuidv4 } from 'uuid';

export default function Home() {
  const [city, setCity] = useState('');
  const [sendKey, setSendKey] = useState('');
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    let uid = localStorage.getItem('weather_uid');
    if (!uid) {
      uid = uuidv4();
      localStorage.setItem('weather_uid', uid);
    }
    const savedKey = localStorage.getItem('weather_sendkey');
    if (savedKey) setSendKey(savedKey);
  }, []);

  const handleSubscribe = async () => {
    setLoading(true);
    try {
      const uid = localStorage.getItem('weather_uid');
      await fetch('/api/subscribe', {
        method: 'POST',
        body: JSON.stringify({
          userId: uid,
          sendKey,
          locationId: '101010100', // 需要替换为实际调用 getCityId 获取的值
          cityName: city || '北京'
        })
      });
      localStorage.setItem('weather_sendkey', sendKey);
      alert('订阅成功！');
    } catch (e) {
      alert('订阅失败');
    }
    setLoading(false);
  };

  return (
    <main className="p-4 max-w-md mx-auto text-black">
      <h1 className="text-2xl font-bold mb-4">极端天气提醒</h1>
      <div className="flex flex-col gap-4">
        <input
          className="border p-2 rounded"
          placeholder="城市 (如: 北京)"
          value={city}
          onChange={e => setCity(e.target.value)}
        />
        <input
          className="border p-2 rounded"
          placeholder="Server酱 SendKey"
          value={sendKey}
          onChange={e => setSendKey(e.target.value)}
        />
        <button
          className="bg-blue-500 text-white p-2 rounded"
          onClick={handleSubscribe}
          disabled={loading}
        >
          {loading ? '保存中...' : '开启提醒'}
        </button>
      </div>
    </main>
  );
}
```
