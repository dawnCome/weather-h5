# H5 适配规范

## 方案：viewport vw/vh

以 375px 为设计稿基准，使用 `vw` 单位适配，配合 Tailwind 自定义工具类。

## viewport 配置

```tsx
// app/layout.tsx
export const viewport: Viewport = {
  width: 'device-width',
  initialScale: 1,
  maximumScale: 1,
  userScalable: false,
  viewportFit: 'cover',   // 支持刘海屏安全区域
}
```

## Tailwind 配置

```ts
// tailwind.config.ts
export default {
  theme: {
    extend: {
      // 设计稿 375px 基准，1vw = 3.75px
      spacing: {
        // 常用尺寸映射：设计稿 px / 3.75 = vw 值
        // 示例：设计稿 16px → spacing-[4.27vw]
      },
      fontSize: {
        'h1': ['6.4vw', { lineHeight: '1.3' }],   // 24px
        'h2': ['5.33vw', { lineHeight: '1.4' }],  // 20px
        'body': ['3.73vw', { lineHeight: '1.6' }], // 14px
        'sm': ['3.2vw', { lineHeight: '1.5' }],   // 12px
      },
      // 最大宽度限制（在平板/PC 上不过宽）
      maxWidth: {
        'mobile': '480px',
      },
    },
  },
}
```

## 使用规范

```tsx
// ✅ 使用 vw 单位或 Tailwind 工具类
<div className="w-full max-w-mobile mx-auto">
  <h1 className="text-h1 font-bold">标题</h1>
  <p className="text-body text-gray-600">正文内容</p>
  <div className="px-[4.27vw] py-[3.2vw]">  {/* 16px / 12px */}
</div>

// ✅ 安全区域适配（刘海屏底部）
<div className="pb-[env(safe-area-inset-bottom)]">
  底部导航
</div>

// ❌ 禁止硬编码 px 做尺寸适配
<div style={{ fontSize: '14px', padding: '16px' }} />
```

## 根布局容器

```tsx
// app/layout.tsx — 统一约束最大宽度，居中显示
export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="zh-CN">
      <body>
        <div className="w-full max-w-mobile mx-auto min-h-screen relative">
          {children}
        </div>
      </body>
    </html>
  )
}
```

## 图片适配

```tsx
// ✅ 使用 next/image，设置 sizes 优化加载
import Image from 'next/image'

<Image
  src="/banner.jpg"
  alt="banner"
  width={750}
  height={400}
  sizes="(max-width: 480px) 100vw, 480px"
  className="w-full h-auto"
/>
```
